Fonts / Typography Configuration

Use separate VS Code font settings for Markdown / LaTeX / TeX so they use serif fonts that are more comfortable for reading.

{
  "editor.fontFamily": "'JetBrains Mono', 'SF Mono', Menlo, monospace",
  "editor.fontSize": 14,
  "editor.lineHeight": 1.45,

  "[markdown]": {
    "editor.fontFamily": "'New York', 'Songti SC', 'LXGW WenKai Screen', serif",
    "editor.fontSize": 16,
    "editor.lineHeight": 1.65,
    "editor.wordWrap": "bounded",
    "editor.wordWrapColumn": 88
  },

  "[latex]": {
    "editor.fontFamily": "'New York', 'Songti SC', 'LXGW WenKai Screen', serif",
    "editor.fontSize": 16,
    "editor.lineHeight": 1.65,
    "editor.wordWrap": "bounded",
    "editor.wordWrapColumn": 88
  },

  "[tex]": {
    "editor.fontFamily": "'New York', 'Songti SC', 'LXGW WenKai Screen', serif",
    "editor.fontSize": 16,
    "editor.lineHeight": 1.65,
    "editor.wordWrap": "bounded",
    "editor.wordWrapColumn": 88
  }
}
